\begin{problema}[40][Armónicos esféricos]
	Considera un sistema físico descrito por el estado cuántico

	\begin{equation*}
		\psi(\theta, \varphi) = \sqrt{\dfrac{3}{8}}Y_{1,1}(\theta,\varphi) +
		\sqrt{\dfrac{1}{8}}Y_{1,0}(\theta, \varphi) +
		AY_{1,-1}(\theta, \varphi),
	\end{equation*}

	donde \(A\) es una constante real.

	\begin{enumerate}
		\item Calcula \(A\), de tal forma que \(\psi\) esté normalizado.
		\item Encuentra \(L_{+}\psi(\theta, \varphi)\).
		\item  Encuentra los valores esperados de \(L_{x}\)
		      y \( \bm{L}^{2}\) en el estado \(\ket{\psi}\).
		\item Encuentra la probabilidad de medir \(\avg{\op{L}{z}} = 0\).
		\item ¿Es \(\psi\) eigenestado de \(\op{L}{\pm}\)?
	\end{enumerate}
\end{problema}
