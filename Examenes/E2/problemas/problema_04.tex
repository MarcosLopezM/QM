\begin{problema}[10][Representación diferencial I]
	Considerando coordenadas eseféricas, donde \(\varphi\) es el ángulo
	azimutal y \(\theta\) el ángulo polar, calcula:

	\begin{enumerate}
		\item \([L_{-}, \mathrm{e}^{-i\varphi} \sin\theta] = 0\).
		\item \([L_{-}, \cos\theta] = \hbar \mathrm{e}^{-i\varphi}\sin\theta\).
	\end{enumerate}
\end{problema}
