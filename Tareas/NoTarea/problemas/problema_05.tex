\begin{problema}
	Justifique el siguiente enunciado

	\begin{equation*}
		\op{J}{+}\ket{j_{1}m_{1}}\ket{j_{2}m_{2}} = \bigl(\op{J}{1+}\ket{j_{1}m_{1}}
		\ket{j_{2}m_{2}}\bigr) + \ket{j_{1}m_{1}} \bigl(\op{J}{2+}\ket{j_{2}m_{2}}\bigr),
	\end{equation*}

	donde \(\op{J}{+} = \op{J}{1+} + \op{J}{2+}\).
\end{problema}
